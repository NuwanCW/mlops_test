27 self, 

28 model: nn.Module, 

29 device: torch.device = torch.device("cpu"),

31 optimizer=None, 

32 scheduler=None, 

33 trial: optuna.trial._trial.Trial = None, 

34 ): 

35 

36 # Set params 

37 self.model = model 

38 self.device = device 

39 self.loss_fn = loss_fn 

40 self.optimizer = optimizer 

41 self.scheduler = scheduler 

42 self.trial = trial

44 def train_step(self, dataloader: torch.utils.data.DataLoader): 

45 """Train step. 

46 

47 Args: 

48 dataloader (torch.utils.data.DataLoader): Torch dataloader to load batches from. 

49 

50 """ 

51 # Set model to train mode 

52 self.model.train() 

53 loss = 0.0 

54 

55 # Iterate over train batches 

56 for i, batch in enumerate(dataloader): 

57 

58 # Step 

59 batch = [item.to(self.device) for item in batch] # Set device 

60 inputs, targets = batch[:-1], batch[-1] 

61 self.optimizer.zero_grad() # Reset gradients 

62 z = self.model(inputs) # Forward pass 

63 J = self.loss_fn(z, targets) # Define loss 

64 J.backward() # Backward pass 

65 self.optimizer.step() # Update weights 

66 

67 # Cumulative Metrics 

68 loss += (J.detach().item() - loss) / (i + 1) 

69 

70 return loss

72 def eval_step(self, dataloader: torch.utils.data.DataLoader): 

73 """Evaluation (val / test) step. 

74 

75 Args: 

76 dataloader (torch.utils.data.DataLoader): Torch dataloader to load batches from. 

77 

78 """ 

79 # Set model to eval mode 

80 self.model.eval() 

81 loss = 0.0 

82 y_trues, y_probs = [], [] 

83 

84 # Iterate over val batches 

85 with torch.inference_mode(): 

86 for i, batch in enumerate(dataloader): 

87 

88 # Step 

89 batch = [item.to(self.device) for item in batch] # Set device 

90 inputs, y_true = batch[:-1], batch[-1] 

91 z = self.model(inputs) # Forward pass 

92 J = self.loss_fn(z, y_true).item() 

93 

94 # Cumulative Metrics 

95 loss += (J - loss) / (i + 1) 

96 

97 # Store outputs 

98 y_prob = torch.sigmoid(z).cpu().numpy() 

99 y_probs.extend(y_prob) 

100 y_trues.extend(y_true.cpu().numpy()) 

101 

102 return loss, np.vstack(y_trues), np.vstack(y_probs)

104 def predict_step(self, dataloader: torch.utils.data.DataLoader): 

105 """Prediction (inference) step. 

106 

107 Note: 

108 Loss is not calculated for this loop. 

109 

110 Args: 

111 dataloader (torch.utils.data.DataLoader): Torch dataloader to load batches from. 

112 

113 """ 

114 # Set model to eval mode 

115 self.model.eval() 

116 y_trues, y_probs = [], [] 

117 

118 # Iterate over batches 

119 with torch.inference_mode(): 

120 for i, batch in enumerate(dataloader): 

121 

122 # Forward pass w/ inputs 

123 batch = [item.to(self.device) for item in batch] # Set device 

124 inputs, y_true = batch[:-1], batch[-1] 

125 z = self.model(inputs) 

126 

127 # Store outputs 

128 y_prob = torch.sigmoid(z).cpu().numpy() 

129 y_probs.extend(y_prob) 

130 y_trues.extend(y_true.cpu().numpy()) 

131 

132 return np.vstack(y_trues), np.vstack(y_probs)

134 def train( 

135 self, 

136 num_epochs: int, 

137 patience: int, 

138 train_dataloader: torch.utils.data.DataLoader, 

139 val_dataloader: torch.utils.data.DataLoader, 

140 ) -> Tuple: 

141 """Training loop. 

142 

143 Args: 

144 num_epochs (int): Maximum number of epochs to train for (can stop earlier based on performance). 

145 patience (int): Number of acceptable epochs for continuous degrading performance. 

146 train_dataloader (torch.utils.data.DataLoader): Dataloader object with training data split. 

147 val_dataloader (torch.utils.data.DataLoader): Dataloader object with validation data split. 

148 

149 Raises: 

150 optuna.TrialPruned: Early stopping of the optimization trial if poor performance. 

151 

152 Returns: 

153 The best validation loss and the trained model from that point. 

154 """ 

155 

156 best_val_loss = np.inf 

157 best_model = None 

158 _patience = patience 

159 for epoch in range(num_epochs): 

160 # Steps 

161 train_loss = self.train_step(dataloader=train_dataloader) 

162 val_loss, _, _ = self.eval_step(dataloader=val_dataloader) 

163 self.scheduler.step(val_loss) 

164 

165 # Pruning based on the intermediate value 

166 if self.trial: 

167 self.trial.report(val_loss, epoch) 

168 if self.trial.should_prune(): # pragma: no cover, optuna pruning 

169 logger.info("Unpromising trial pruned!") 

170 raise optuna.TrialPruned() 

171 

172 # Early stopping 

173 if val_loss < best_val_loss: 

174 best_val_loss = val_loss 

175 best_model = self.model 

176 _patience = patience # reset _patience 

177 else: # pragma: no cover, simple subtraction 

178 _patience -= 1 

179 if not _patience: # pragma: no cover, simple break 

180 logger.info("Stopping early!") 

181 break 

182 

183 # Logging 

184 logger.info( 

185 f"Epoch: {epoch+1} | " 

186 f"train_loss: {train_loss:.5f}, " 

187 f"val_loss: {val_loss:.5f}, " 

188 f"lr: {self.optimizer.param_groups[0]['lr']:.2E}, " 

189 f"_patience: {_patience}" 

190 ) 

191 return best_val_loss, best_model

194def find_best_threshold(y_true: np.ndarray, y_prob: np.ndarray) -> float: 

195 """Determine the best threshold for maximum f1 score. 

196 

197 Usage: 

198 

199 ```python 

200 # Find best threshold 

201 _, y_true, y_prob = trainer.eval_step(dataloader=train_dataloader) 

202 params.threshold = find_best_threshold(y_true=y_true, y_prob=y_prob) 

203 ``` 

204 

205 Args: 

206 y_true (np.ndarray): True labels. 

207 y_prob (np.ndarray): Probability distribution for predicted labels. 

208 

209 Returns: 

210 Best threshold for maximum f1 score. 

211 """ 

212 precisions, recalls, thresholds = precision_recall_curve(y_true.ravel(), y_prob.ravel()) 

213 f1s = (2 * precisions * recalls) / (precisions + recalls) 

214 return thresholds[np.argmax(f1s)]

217def train(params: Namespace, trial: optuna.trial._trial.Trial = None) -> Dict: 

218 """Operations for training. 

219 

220 Args: 

221 params (Namespace): Input parameters for operations. 

222 trial (optuna.trial._trial.Trail, optional): Optuna optimization trial. Defaults to None. 

223 

224 Returns: 

225 Artifacts to save and load for later. 

226 """ 

227 # Set up 

228 utils.set_seed(seed=params.seed) 

229 device = utils.set_device(cuda=params.cuda) 

230 

231 # Load features 

232 features_fp = Path(config.DATA_DIR, "features.json") 

233 tags_fp = Path(config.DATA_DIR, "tags.json") 

234 features = utils.load_dict(filepath=features_fp) 

235 tags_dict = utils.list_to_dict(utils.load_dict(filepath=tags_fp), key="tag") 

236 df = pd.DataFrame(features) 

237 if params.shuffle: 

238 df = df.sample(frac=1).reset_index(drop=True) 

239 df = df[: params.subset] # None = all samples 

240 

241 # Prepare data (filter, clean, etc.) 

242 df, tags_above_freq, tags_below_freq = data.prepare( 

243 df=df, 

244 include=list(tags_dict.keys()), 

245 exclude=config.EXCLUDED_TAGS, 

246 min_tag_freq=params.min_tag_freq, 

247 ) 

248 params.num_samples = len(df) 

249 

250 # Preprocess data 

251 df.text = df.text.apply(data.preprocess, lower=params.lower, stem=params.stem) 

252 

253 # Encode labels 

254 labels = df.tags 

255 label_encoder = data.MultiLabelLabelEncoder() 

256 label_encoder.fit(labels) 

257 y = label_encoder.encode(labels) 

258 

259 # Class weights 

260 all_tags = list(itertools.chain.from_iterable(labels.values)) 

261 counts = np.bincount([label_encoder.class_to_index[class_] for class_ in all_tags]) 

262 class_weights = {i: 1.0 / count for i, count in enumerate(counts)} 

263 

264 # Split data 

265 utils.set_seed(seed=params.seed) # needed for skmultilearn 

266 X = df.text.to_numpy() 

267 X_train, X_, y_train, y_ = data.iterative_train_test_split( 

268 X=X, y=y, train_size=params.train_size 

269 ) 

270 X_val, X_test, y_val, y_test = data.iterative_train_test_split(X=X_, y=y_, train_size=0.5) 

271 test_df = pd.DataFrame({"text": X_test, "tags": label_encoder.decode(y_test)}) 

272 

273 # Tokenize inputs 

274 tokenizer = data.Tokenizer(char_level=params.char_level) 

275 tokenizer.fit_on_texts(texts=X_train) 

276 X_train = np.array(tokenizer.texts_to_sequences(X_train), dtype=object) 

277 X_val = np.array(tokenizer.texts_to_sequences(X_val), dtype=object) 

278 X_test = np.array(tokenizer.texts_to_sequences(X_test), dtype=object) 

279 

280 # Create dataloaders 

281 train_dataset = data.CNNTextDataset( 

282 X=X_train, y=y_train, max_filter_size=params.max_filter_size 

283 ) 

284 val_dataset = data.CNNTextDataset(X=X_val, y=y_val, max_filter_size=params.max_filter_size) 

285 train_dataloader = train_dataset.create_dataloader(batch_size=params.batch_size) 

286 val_dataloader = val_dataset.create_dataloader(batch_size=params.batch_size) 

287 

288 # Initialize model 

289 model = models.initialize_model( 

290 params=params, 

291 vocab_size=len(tokenizer), 

292 num_classes=len(label_encoder), 

293 device=device, 

294 ) 

295 

296 # Train model 

297 logger.info(f"Parameters: {json.dumps(params.__dict__, indent=2, cls=NumpyEncoder)}") 

298 class_weights_tensor = torch.Tensor(np.array(list(class_weights.values()))) 

299 loss_fn = nn.BCEWithLogitsLoss(weight=class_weights_tensor) 

300 optimizer = torch.optim.Adam(model.parameters(), lr=params.lr) 

301 scheduler = torch.optim.lr_scheduler.ReduceLROnPlateau( 

302 optimizer, mode="min", factor=0.05, patience=5 

303 ) 

304 

305 # Trainer module 

306 trainer = Trainer( 

307 model=model, 

308 device=device, 

309 loss_fn=loss_fn, 

310 optimizer=optimizer, 

311 scheduler=scheduler, 

312 trial=trial, 

313 ) 

314 

315 # Train 

316 best_val_loss, best_model = trainer.train( 

317 params.num_epochs, params.patience, train_dataloader, val_dataloader 

318 ) 

319 

320 # Find best threshold 

321 _, y_true, y_prob = trainer.eval_step(dataloader=train_dataloader) 

322 params.threshold = find_best_threshold(y_true=y_true, y_prob=y_prob) 

323 

324 # Evaluate model 

325 artifacts = { 

326 "params": params, 

327 "label_encoder": label_encoder, 

328 "tokenizer": tokenizer, 

329 "model": best_model, 

330 "loss": best_val_loss, 

331 } 

332 device = torch.device("cpu") 

333 y_true, y_pred, performance = eval.evaluate(df=test_df, artifacts=artifacts) 

334 artifacts["performance"] = performance 

335 

336 return artifacts

339def objective(params: Namespace, trial: optuna.trial._trial.Trial) -> float: 

340 """Objective function for optimization trials. 

341 

342 Args: 

343 params (Namespace): Input parameters for each trial (see `config/params.json`). 

344 trial (optuna.trial._trial.Trial): Optuna optimization trial. 

345 

346 Returns: 

347 F1 score from evaluating the trained model on the test data split. 

348 """ 

349 # Paramters (to tune) 

350 params.embedding_dim = trial.suggest_int("embedding_dim", 128, 512) 

351 params.num_filters = trial.suggest_int("num_filters", 128, 512) 

352 params.hidden_dim = trial.suggest_int("hidden_dim", 128, 512) 

353 params.dropout_p = trial.suggest_uniform("dropout_p", 0.3, 0.8) 

354 params.lr = trial.suggest_loguniform("lr", 5e-5, 5e-4) 

355 

356 # Train (can move some of these outside for efficiency) 

357 logger.info(f"\nTrial {trial.number}:") 

358 logger.info(json.dumps(trial.params, indent=2)) 

359 artifacts = train(params=params, trial=trial) 

360 

361 # Set additional attributes 

362 params = artifacts["params"] 

363 performance = artifacts["performance"] 

364 logger.info(json.dumps(performance["overall"], indent=2)) 

365 trial.set_user_attr("threshold", params.threshold) 

366 trial.set_user_attr("precision", performance["overall"]["precision"]) 

367 trial.set_user_attr("recall", performance["overall"]["recall"]) 

368 trial.set_user_attr("f1", performance["overall"]["f1"]) 

369 

370 return performance["overall"]["f1"]
