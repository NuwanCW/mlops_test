56def list_to_dict(list_of_dicts: List, key: str) -> Dict: 

57 """Convert a list of `dict_a` to a `dict_b` where 

58 the `key` in `dict_b` is an item in each `dict_a`. 

59 

60 Args: 

61 list_of_dicts (List): list of items to convert to dict. 

62 key (str): Name of the item in `dict_a` to use as primary key for `dict_b`. 

63 

64 Returns: 

65 A dictionary with items from the list organized by key. 

66 """ 

67 d_b = {} 

68 for d_a in list_of_dicts: 

69 d_b_key = d_a.pop(key) 

70 d_b[d_b_key] = d_a 

71 return d_b
